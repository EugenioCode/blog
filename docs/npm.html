    NPM

    jianjun.yang 3/21/2022 NpmNode

    # 列举一些实用的NPM包

    # 1. 切换npm源的工具 —— nrm

    # 1.1 nrm的安装

    npm install -g nrm
    1

    # 1.2 查看可选npm源

    nrm ls
    1

    # 1.3 切换npm源

    nrm use npm
    1

    # 1.4 增加npm源

    nrm add <name> <url>
    1

    # 1.5 删除npm源

    nrm del <name>
    1

    # 1.6 测试npm源速度

    nrm test [<name>]
    1

    # 2. 切换node版本的工具 —— n

    # 2.1 安装 n

    npm install -g n
    1

    # 2.2 查看可用的 node 版本

    n
    1

    # 2.3 基本命令

    // 稳定版本
n stable
 
// 最新版
n lastest
 
// 指定版本
n 16.15.1

// 删除指定版本
sudo n rm 16.15.1
